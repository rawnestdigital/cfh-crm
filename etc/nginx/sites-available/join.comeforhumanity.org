# Production Nginx Configuration
# Join Come for Humanity CRM

# Rate limiting zones
limit_req_zone $binary_remote_addr zone=login:10m rate=5r/m;
limit_req_zone $binary_remote_addr zone=api:10m rate=60r/m;
limit_req_zone $binary_remote_addr zone=general:10m rate=100r/m;

# Upstream PHP-FPM
upstream php_backend {
    server unix:/run/php/php8.1-fpm.sock;
}

# HTTP to HTTPS redirect
server {
    listen 80;
    listen [::]:80;
    server_name join.comeforhumanity.org;

    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }

    location / {
        return 301 https://$server_name$request_uri;
    }
}

# HTTPS Server
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name join.comeforhumanity.org;

    # SSL Certificates
    ssl_certificate /etc/letsencrypt/live/join.comeforhumanity.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/join.comeforhumanity.org/privkey.pem;
    ssl_trusted_certificate /etc/letsencrypt/live/join.comeforhumanity.org/chain.pem;

    # SSL Configuration
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers on;
    ssl_session_cache shared:SSL:50m;
    ssl_session_timeout 1d;
    ssl_session_tickets off;

    # OCSP Stapling
    ssl_stapling on;
    ssl_stapling_verify on;
    ssl_stapling_file /var/run/ocsp.staple;

    # Security Headers
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;
    add_header X-Frame-Options "DENY" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Permissions-Policy "geolocation=(), microphone=(), camera=()" always;
    add_header Content-Security-Policy "default-src 'self'; script-src 'self'; style-src 'self' 'unsafe-inline'; img-src 'self' data: https:;" always;

    # Root directory
    root /var/www/cfh-crm/public;
    index index.php index.html;

    # Logging
    access_log /var/log/nginx/cfh-crm-access.log combined buffer=32k flush=1m;
    error_log /var/log/nginx/cfh-crm-error.log warn;

    # Client upload limit
    client_max_body_size 5M;
    client_body_timeout 30s;
    client_header_timeout 30s;
    keepalive_timeout 30s;

    # Hide version info
    server_tokens off;

    # Block access to hidden files
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }

    # Block access to config files
    location ~ (^|/)config {
        deny all;
    }

    # Block access to storage
    location ~ ^/storage {
        deny all;
    }

    # Block access to backup files
    location ~ ~$ {
        deny all;
    }

    # Deny access to PHP files outside public directory
    location ~ ^/(?!public/) {
        deny all;
    }

    # Static assets - cache aggressively
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|svg|woff|woff2|ttf|eot|otf)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
        add_header X-Content-Type-Options "nosniff";
        access_log off;
    }

    # PHP-FPM
    location ~ \.php$ {
        try_files $uri =404;
        fastcgi_pass php_backend;
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        fastcgi_param SCRIPT_NAME $fastcgi_script_name;

        # FastCGI parameters
        include fastcgi_params;

        # Security parameters
        fastcgi_param HTTPS on;
        fastcgi_param HTTP_SCHEME https;
        fastcgi_param SERVER_PORT 443;

        # Timeouts
        fastcgi_connect_timeout 30s;
        fastcgi_send_timeout 30s;
        fastcgi_read_timeout 30s;
    }

    # API endpoints
    location /api/ {
        limit_req zone=api burst=10 nodelay;
        try_files $uri $uri/ /index.php?$query_string;
    }

    # Login endpoint - strict rate limiting
    location /login {
        limit_req zone=login burst=2 nodelay;
        try_files $uri $uri/ /index.php?$query_string;
    }

    # General requests
    location / {
        limit_req zone=general burst=20 nodelay;
        try_files $uri $uri/ /index.php?$query_string;
    }

    # Deny access to uploads from web
    location /secure_uploads {
        deny all;
    }

    # Health check endpoint
    location /health {
        access_log off;
        return 200 "OK";
        add_header Content-Type text/plain;
    }
}
